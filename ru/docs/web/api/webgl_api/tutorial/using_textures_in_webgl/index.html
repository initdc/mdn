function loadTexture(gl, url) {
  const texture = gl.createTexture();
  gl.bindTexture(gl.TEXTURE_2D, texture);

  // –¢–∞–∫ –∫–∞–∫ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–µ –±—É–¥–µ—Ç –∑–∞–≥—Ä—É–∂–µ–Ω–æ –∏–∑ –∏–Ω—Ç–µ—Ä–Ω–µ—Ç–∞,
  // –º–æ–∂–µ—Ç –ø–æ—Ç—Ä–µ–±–æ–≤–∞—Ç—å—Å—è –≤—Ä–µ–º—è –¥–ª—è –ø–æ–ª–Ω–æ–π –∑–∞–≥—Ä—É–∑–∫–∏.
  // –ü–æ—ç—Ç–æ–º—É —Å–Ω–∞—á–∞–ª–∞ –º—ã –ø–æ–º–µ—â–∞–µ–º –≤ —Ç–µ–∫—Å—Ç—É—Ä—É –µ–¥–∏–Ω—Å—Ç–≤–µ–Ω–Ω—ã–π –ø–∏–∫—Å–µ–ª—å, —á—Ç–æ–±—ã
  // –µ—ë –º–æ–∂–Ω–æ –±—ã–ª–æ –∏—Å–ø–æ–ª—å–∑–æ–≤–∞—Ç—å —Å—Ä–∞–∑—É. –ü–æ—Å–ª–µ –∑–∞–≤–µ—Ä—à–µ–Ω–∏—è –∑–∞–≥—Ä—É–∑–∫–∏
  // –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è –º—ã –æ–±–Ω–æ–≤–∏–º —Ç–µ–∫—Å—Ç—É—Ä—É.
  const level = 0;
  const internalFormat = gl.RGBA;
  const width = 1;
  const height = 1;
  const border = 0;
  const srcFormat = gl.RGBA;
  const srcType = gl.UNSIGNED_BYTE;
  const pixel = new Uint8Array([0, 0, 255, 255]);  // –Ω–µ–ø—Ä–æ–∑—Ä–∞—á–Ω—ã–π —Å–∏–Ω–∏–π
  gl.texImage2D(gl.TEXTURE_2D, level, internalFormat,
                width, height, border, srcFormat, srcType,
                pixel);

  const image = new Image();

    gl.bindTexture(gl.TEXTURE_2D, texture);
    gl.texImage2D(gl.TEXTURE_2D, level, internalFormat,
                  srcFormat, srcType, image);

    // –£ WebGL1 –∏–Ω—ã–µ —Ç—Ä–µ–±–æ–≤–∞–Ω–∏—è –∫ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è–º, –∏–º–µ—é—â–∏–º —Ä–∞–∑–º–µ—Ä —Å—Ç–µ–ø–µ–Ω–∏ 2,
    // –∏ –∫ –Ω–µ –∏–º–µ—é—â–∏–º —Ä–∞–∑–º–µ—Ä —Å—Ç–µ–ø–µ–Ω–∏ 2, –ø–æ—ç—Ç–æ–º—É –ø—Ä–æ–≤–µ—Ä—è–µ–º, —á—Ç–æ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–µ
    // –∏–º–µ–µ—Ç —Ä–∞–∑–º–µ—Ä —Å—Ç–µ–ø–µ–Ω–∏ 2 –≤ –æ–±–µ–∏—Ö –∏–∑–º–µ—Ä–µ–Ω–∏—è—Ö.

       // –†–∞–∑–º–µ—Ä —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤—É–µ—Ç —Å—Ç–µ–ø–µ–Ω–∏ 2. –°–æ–∑–¥–∞—ë–º MIP'—ã.
       gl.generateMipmap(gl.TEXTURE_2D);

       // –†–∞–∑–º–µ—Ä –Ω–µ —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤—É–µ—Ç —Å—Ç–µ–ø–µ–Ω–∏ 2.
       // –û—Ç–∫–ª—é—á–∞–µ–º MIP'—ã –∏ —É—Å—Ç–∞–Ω–∞–≤–ª–∏–≤–∞–µ–º –Ω–∞—Ç—è–∂–µ–Ω–∏–µ –ø–æ –∫—Ä–∞—è–º
       gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_S, gl.CLAMP_TO_EDGE);
       gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_WRAP_T, gl.CLAMP_TO_EDGE);
       gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.LINEAR);
    }
  };

  return texture;
}

function isPowerOf2(value) {
  return (value & (value - 1)) == 0;
}
